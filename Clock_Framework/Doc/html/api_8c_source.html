291unsigned int getAdcResult(unsigned char ch)
292{
293 ADC_CONTR = ADC_POWER | ADC_SPEEDLL | ch | ADC_START;
294 _nop_(); // Must wait before inquiry
295 _nop_();
296 _nop_();
297 _nop_();
298 _nop_(); // Must wait before inquiry
299 _nop_();
300 while (!(ADC_CONTR & ADC_FLAG))
301 ; // Wait complete flag
302 ADC_CONTR &= ~ADC_FLAG; // Close ADC
303
304 return (ADC_RES * 4 + ADC_RESL); // Return ADC result.
305}
